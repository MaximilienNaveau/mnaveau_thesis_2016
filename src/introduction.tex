        [Kuindersma Autonomous Robots 2015]

      \end{alertblock}
      \vspace*{-0.8cm}
    }
    \only<3>{
      \begin{alertblock}{3D PG}
        [Hirukawa ICRA 2007]
        
        Contacts and Forces Distribution Known
        
        [Perrin IJRR 2015]
        
        $\mathbf{\dot{L}}=0$ ; Forces Direction Known
      \end{alertblock}
      \vspace*{-0.8cm}
      \begin{align*}
        m\,( \mathbf{ \ddot{c} } - \mathbf{g}) &= \sum_{i} \mathbf{R} _{i} \mathbf{f} _{i}\\
        m \,\mathbf{c} \times ( \mathbf {\ddot{c}} - \mathbf{g} ) + \mathbf{\dot{L}} &= \sum_{i} \mathbf{p} _{i} \times \mathbf{R} _{i} \mathbf{f} _{i}\\
        \mathcal{A}({\bf q},{\bf \lambda}) \geq 0 \iff & \mu f^2_z > f^2_x + f^2_y
      \end{align*}    
    }
    \only<4>{
      \begin{alertblock}{2D PG}
        [Herdt Advanced Robotics 2010]        
        [Sherikov Humnaoids 2014]
      \end{alertblock}
      \vspace*{-0.3cm}
      \begin{alertblock}{Linearized Inverted Pendulum Model}
        \itemd coplanar contact
      
        \itemd $\dot{\bf \am} = 0$ \itemd $c^z = cst$ \itemd ${\bf p}=CoP$
      \end{alertblock}
      \vspace*{-0.5cm}
      \begin{equation*}
        \left\lbrace
        \begin{aligned}
          p_{x} = c_x - \frac{c_z}{g}\ddot{c}_x \\
          p_{y} = c_y - \frac{c_z}{g}\ddot{c}_y \\
          {\bf p} \in \text{Support Polygon}
        \end{aligned}
        \right.
      \end{equation*}
    }
